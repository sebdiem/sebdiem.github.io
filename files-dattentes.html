def init():
    vehicles.set_data([], [])
    return vehicles, 

def anim(i):
    data = []
    for t in times:
        if i > t:
            if data:
                if i-t < (data[-1] - 1):
                    data.append(i-t)
                else: data.append(data[-1] - 0.5)
            else:
                if i-t < 50: data.append(i-t)
                else: data.append(50)
    vehicles.set_data(data, 0.5*np.ones(len(data)))
    return vehicles,

anim = animation.FuncAnimation(fig, anim, init_func=init, frames=1000, interval=30, blit=True)
